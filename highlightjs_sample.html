C# Server Algorithm with Infinite Loop

Example showing a typical server loop pattern - the infinite loop section is highlighted in red

using System;
using System.Threading;

namespace ServerExample
{
    class ServerProgram
    {
        static void Main(string[] args)
        {
            Console.WriteLine("Server starting...");
            bool isRunning = true;
            int requestCount = 0;
            
            // Initialize server
            InitializeServer();
            
            // Main server loop - runs indefinitely
            while (isRunning)
            {
                // Wait for incoming request
                var request = WaitForRequest();
                
                if (request != null)
                {
                    requestCount++;
                    ProcessRequest(request);
                    Console.WriteLine($"Processed request #{requestCount}");
                }
                
                // Small delay to prevent CPU overuse
                Thread.Sleep(100);
            }
            
            Console.WriteLine("Server stopped.");
        }
        
        static void InitializeServer()
        {
            Console.WriteLine("Server initialized on port 8080");
        }
        
        static object WaitForRequest()
        {
            // Simulate request handling
            return new object();
        }
        
        static void ProcessRequest(object request)
        {
            // Process the request
            Thread.Sleep(50);
        }
    }
}
